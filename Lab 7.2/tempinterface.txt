	/************************************************************************
Avalon-MM Interface VGA Text mode display

Register Map:
0x000-0x0257 : VRAM, 80x30 (2400 byte, 600 word) raster order (first column then row)
0x258        : control register

VRAM Format:
X->
[ 31  30-24][ 23  22-16][ 15  14-8 ][ 7    6-0 ]
[IV3][CODE3][IV2][CODE2][IV1][CODE1][IV0][CODE0]

IVn = Draw inverse glyph
CODEn = Glyph code from IBM codepage 437

Control Register Format:
[[31-25][24-21][20-17][16-13][ 12-9][ 8-5 ][ 4-1 ][   0    ] 
[[RSVD ][FGD_R][FGD_G][FGD_B][BKG_R][BKG_G][BKG_B][RESERVED]

VSYNC signal = bit which flips on every Vsync (time for new frame), used to synchronize software
BKG_R/G/B = Background color, flipped with foreground when IVn bit is set
FGD_R/G/B = Foreground color, flipped with background when Inv bit is set

************************************************************************/

									.rden_b(1'b1), //always high?

				4'b0001 : color_palette[AVL_ADDR[3:0]][7:0] <= AVL_WRITEDATA[7:0];

				4'b0100 : color_palette[AVL_ADDR[3:0]][23:16] <= AVL_WRITEDATA[23:16];
				4'b1000 : color_palette[AVL_ADDR[3:0]][31:24] <= AVL_WRITEDATA[31:24];
				4'b0011 : color_palette[AVL_ADDR[3:0]][15:0] <= AVL_WRITEDATA[15:0];
				4'b1100 : color_palette[AVL_ADDR[3:0]][31:16] <= AVL_WRITEDATA[31:16];
				4'b1111 : color_palette[AVL_ADDR[3:0]][31:0] <= AVL_WRITEDATA[31:0];
	
				default: ;
			endcase
			
		end
	else
		begin
		end
end
